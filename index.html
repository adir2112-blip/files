// Supabase Edge Function — wa-alerts // Schedule: every 2 hours (or as configured) // Env vars needed: SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' const WA_API = 'https://whatsapp-bot-production-120a.up.railway.app/api/send' async function sendWA(phone: string, message: string): Promise { try { const res = await fetch(WA_API, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ phone, message }) }) return res.ok } catch (e) { console.error('WA send failed:', e) return false } } function formatPhone(phone: string): string { const digits = phone.replace(/[^0-9]/g, '') if (digits.startsWith('972')) return digits if (digits.startsWith('0')) return '972' + digits.slice(1) return '972' + digits } function businessDaysBetween(from: Date, to: Date): number { let count = 0 const cur = new Date(from) cur.setHours(0, 0, 0, 0) const end = new Date(to) end.setHours(0, 0, 0, 0) while (cur < end) { const day = cur.getDay() if (day !== 5 && day !== 6) count++ cur.setDate(cur.getDate() + 1) } return count } function isAllowedNow(policy: any): boolean { if (!policy?.enabled) return false const now = new Date() const day = now.getDay() if ((policy.blockedDays || []).includes(day)) return false const hour = now.getHours() if (hour < (policy.startHour ?? 8) || hour >= (policy.endHour ?? 20)) return false return true } function recordMatchesAlert(record: any, file: any): boolean { const rule = file.alert_rule || {} if (!rule.enabled || !rule.fieldId) return false if (rule.days) { const age = businessDaysBetween(new Date(record.created_at), new Date()) if (age < parseFloat(rule.days)) return false } const adminFields = file.admin_fields || [] const isAdmin = adminFields.find((f: any) => f.id === rule.fieldId) const val = isAdmin ? ((record.admin_data || {})[rule.fieldId] || '').toString() : ((record.extra || {})[rule.fieldId] || '').toString() if (rule.condition === 'empty') return !val || val.trim() === '' if (rule.condition === 'not_value') return val !== (rule.value || '') if (rule.condition === 'value') return val === (rule.value || '') return false } Deno.serve(async (req) => { try { const SUPABASE_URL = Deno.env.get('SUPABASE_URL')! const SUPABASE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')! const sb = createClient(SUPABASE_URL, SUPABASE_KEY) // Load files + orgs const { data: files } = await sb.from('files').select('*') const { data: orgs } = await sb.from('organizations').select('*') if (!files?.length) { return new Response(JSON.stringify({ ok: true, msg: 'no files' })) } // Check if WA alerts are enabled via Supabase settings const { data: settingRow } = await sb .from('app_settings') .select('value') .eq('key', 'wa_enabled') .single() const waEnabled = settingRow ? JSON.parse(settingRow.value) : true if (!waEnabled) { return new Response(JSON.stringify({ ok: true, msg: 'WA alerts disabled by admin' }), { headers: { 'Content-Type': 'application/json' } }) } // Default policy — hours in UTC (Israel = UTC+3 in summer) // 08:00 Israel = 05:00 UTC, 20:00 Israel = 17:00 UTC const defaultPolicy = { enabled: true, startHour: 5, // 08:00 Israel time endHour: 17, // 20:00 Israel time blockedDays: [5, 6] } if (!isAllowedNow(defaultPolicy)) { const now = new Date() return new Response(JSON.stringify({ ok: true, msg: `outside window — day ${now.getDay()} hour ${now.getHours()}` })) } const filesWithAlerts = files.filter((f: any) => f.alert_rule?.enabled && f.alert_rule?.fieldId && (f.wa_contacts || []).length > 0 ) if (!filesWithAlerts.length) { return new Response(JSON.stringify({ ok: true, msg: 'no files with WA+SLA' })) } let totalSent = 0 const log: any[] = [] for (const f of filesWithAlerts) { const { data: recs } = await sb .from('records') .select('id,created_at,extra,admin_data') .eq('file_id', f.id) .eq('archived', false) const breached = (recs || []).filter((r: any) => recordMatchesAlert(r, f)) if (!breached.length) continue const org = (orgs || []).find((o: any) => o.id === f.org_id) const allFields = [...(f.extra_fields || []), ...(f.admin_fields || [])] const rule = f.alert_rule || {} const fieldLabel = (allFields.find((x: any) => x.id === rule.fieldId) || {}).label || rule.fieldId const dateStr = new Date().toLocaleDateString('he-IL') const msg = `עדכון מובמנט - ${dateStr}\n\n${f.name}: ${breached.length} ממתינות\n\nלכניסה למערכת: https://movement-moked.github.io` for (const contact of (f.wa_contacts || [])) { const phone = formatPhone(contact.phone) const ok = await sendWA(phone, msg) totalSent++ log.push({ file: f.name, contact: contact.name, phone, breached: breached.length, ok }) } } return new Response(JSON.stringify({ ok: true, totalSent, log }), { headers: { 'Content-Type': 'application/json' } }) } catch (err) { return new Response(JSON.stringify({ ok: false, error: String(err) }), { status: 500, headers: { 'Content-Type': 'application/json' } }) } })
